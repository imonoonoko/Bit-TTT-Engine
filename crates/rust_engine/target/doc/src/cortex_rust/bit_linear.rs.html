33    /// Forward pass with quantization
34    /// x: Input vector (f32)
35    /// Returns: Output vector (f32)
36    pub fn forward(&self, x: &Array1<f32>) -> Array1<f32> {
37        // 1. Quantize Input to i8 (Range [-127, 127])
38        // Simplified scale: max abs value mapping to 127
39        let max_val = x.fold(0.0f32, |a, &b| a.max(b.abs()));
40        let scale = if max_val > 1e-9 { 127.0 / max_val } else { 1.0 };
41        
42        let x_quant: Array1<i8> = x.mapv(|v| (v * scale).round().clamp(-127.0, 127.0) as i8);
43        
44        // 2. Integer Matrix Multiplication (i8 * i8 -> i32)
45        // Manual loop implementation for clarity/no-blas dependency for now
46        let mut y_int = Array1::<i32>::zeros(self.out_features);
47        
48        for out_idx in 0..self.out_features {
49            let mut sum: i32 = 0;
50            for in_idx in 0..self.in_features {
51                let w = self.weights[[in_idx, out_idx]] as i32;
52                let inp = x_quant[in_idx] as i32;
53                sum += w * inp;
54            }
55            y_int[out_idx] = sum;
56        }
57        
58        // 3. Dequantize
59        // y_float = y_int / scale
60        // Note: Weights scale is 1.0 (since they are pure {-1, 0, 1})
61        // Real BitNet has a weight scale too, but let's ignore for this prototype loop.
62        
63        let y = y_int.mapv(|v| (v as f32) / scale);
